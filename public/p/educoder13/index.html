def feature_map_vis(model, image):
    '''
    Â∞ÜmodelÁöÑÁ¨¨6Â±ÇÊâÄÁîüÊàêÊúÄÂêé4‰∏™ÁâπÂæÅÂõæÂèØËßÜÂåñÔºåÂπ∂‰øùÂ≠òÂà∞./step1/result/feature_map.png‰∏≠
    :param model: VGG16Ê®°Âûã

    #********* Begin *********#

        plt.imshow(show_img, cmap='gray')
        plt.axis('off')
    plt.savefig('./step1/result/feature_map.png')

from keras import backend as K
import numpy as np


def update_img(epochs, input_layer, feature_map, learning_rate):
    '''
    ÈöèÊú∫ÁîüÊàê1Âº†ÂÆΩ224ÔºåÈ´ò224ÁöÑRGBÂõæÂÉèÔºåÂπ∂Êõ¥Ë°åÂõæÂÉè
    :param epochs: Ê¢ØÂ∫¶‰∏äÂçáÁöÑËø≠‰ª£Ê¨°Êï∞
    :param input_layer: CNNÊ®°ÂûãÁöÑËæìÂÖ•Â±Ç
    :param feature_map: CNNÊ®°ÂûãÁöÑÊåáÂÆöÁöÑÁâπÂæÅÂõæ
    :param learning_rate: Ê¢ØÂ∫¶‰∏äÂçáÁöÑÂ≠¶‰π†Áéá
    :return: Êõ¥Êñ∞ÂêéÁöÑÂõæÂÉèÂíåÔºåÂÖ∂‰∏≠ÂõæÂÉèÁöÑÁª¥Â∫¶‰∏∫(ÂõæÂÉèÁöÑÈ´òÔºåÂõæÂÉèÁöÑÂÆΩÔºåÈÄöÈÅìÊï∞)
    '''

    #********* Begin *********#
    # ÂàõÂª∫‰∏Ä‰∏™ÈöèÊú∫ÂàùÂßãÂåñÁöÑÂõæÂÉè
    img = np.random.randint(low=0, high=255, size=(1, 224, 224, 3))
    img = img / 255.0  # ÂΩí‰∏ÄÂåñÂ§ÑÁêÜ

    # ÂÆö‰πâÊçüÂ§±ÂáΩÊï∞‰∏∫ÁâπÂæÅÂõæÁöÑÂùáÂÄº
    loss = K.mean(feature_map)

    # ËÆ°ÁÆóÊçüÂ§±ÂáΩÊï∞ÂØπÂõæÂÉèÁöÑÊ¢ØÂ∫¶Âπ∂ËøõË°åÂΩí‰∏ÄÂåñ
    grads = K.gradients(loss, input_layer)[0]
    grads = grads / (K.sqrt(K.mean(K.square(grads))) + K.epsilon())

    # ÂÆö‰πâÂâçÂêë‰º†Êí≠Êìç‰Ωú
    iterate = K.function([input_layer], [loss, grads])

    # Ëø≠‰ª£epochsÊ¨°Ê¢ØÂ∫¶‰∏äÂçá
    for _ in range(epochs):
        # ÂæóÂà∞ÂΩìÂâçÁöÑÊçüÂ§±ÂÄº‰∏éÊ¢ØÂ∫¶
        loss_value, grads_value = iterate([img])
        # Ê†πÊçÆÊ¢ØÂ∫¶‰∏äÂçáÁöÑÂÖ¨ÂºèÊõ¥Êñ∞ÂõæÂÉè
        img += grads_value * learning_rate
        # Â¶ÇÊûúÊõ¥Êñ∞‰∏çÂä®‰∫ÜÂ∞±ÊèêÂâçËøîÂõû
        if loss_value <= K.epsilon():
            break

    # ËøîÂõûÊõ¥Êñ∞ÂêéÁöÑÂõæÂÉè
    return img[0]
    #********* End *********#

from keras import backend as K
import numpy as np
def deep_dream(img, model, epochs, learning_rate, max_threshold):
    '''
    Â∞ÜVGG16ÁöÑblock4_conv1‰∏éblock5_conv2Áî®Êù•ÂÅödeep dreamÔºåÂÖ∂‰∏≠block4_conv1ÁöÑÊùÉÈáç‰∏∫0.05Ôºåblock5_conv2ÁöÑÊùÉÈáç‰∏∫0.2ÔºåÂπ∂Â∞ÜÂõæÂÉèËøîÂõû
    :param img: È¢ÑÂ§ÑÁêÜÂêéÁöÑÂõæÂÉèÔºåÁ±ªÂûã‰∏∫ndarrayÔºåshape‰∏∫(Ê†∑Êú¨‰∏™Êï∞ÔºåÂõæÂÉèÁöÑÂÆΩÔºåÂõæÂÉèÁöÑÈ´òÔºåÈÄöÈÅìÊï∞)
    :param model: VGG16
    :param epochs: Ê¢ØÂ∫¶‰∏äÂçáÁöÑËø≠‰ª£Ê¨°Êï∞
    :param learning_rate: Ê¢ØÂ∫¶‰∏äÂçáÁöÑÂ≠¶‰π†Áéá
    :param max_threshold: ÊçüÂ§±ÂÄºÁöÑÊúÄÂ§ßÈòàÂÄº
    :return: deep dreamÂêéÁöÑÂõæÂÉèÔºåÂÖ∂‰∏≠ÂõæÂÉèÁöÑÁª¥Â∫¶‰∏∫(ÂõæÂÉèÁöÑÈ´òÔºåÂõæÂÉèÁöÑÂÆΩÔºåÈÄöÈÅìÊï∞)
    '''
    #********* Begin *********#
    layer_contributions = {
        'block4_conv1': 0.05,
        'block5_conv2': 0.2
    }
    # ÂÆö‰πâ‰∏Ä‰∏™ÂåÖÂê´ÊçüÂ§±ÁöÑÂº†ÈáèÔºåÂç≥‰∏äÈù¢ÂàóÂá∫ÁöÑÊøÄÊ¥ªÁöÑL2ËåÉÊï∞ÁöÑÂä†ÊùÉÂíå„ÄÇ
    layer_dict = dict([(layer.name, layer) for layer in model.layers])
    # ÂÆö‰πâÊçüÂ§±
    loss = K.variable(0.)
    for layer_name in layer_contributions:
        # Â∞Ü‰∏¢Â§±ÁöÑÂõæÂ±ÇÁâπÂæÅÁöÑL2ËåÉÊï∞Ê∑ªÂä†Âà∞ÊçüÂ§±‰∏≠„ÄÇ
        coeff = layer_contributions[layer_name]
        activation = layer_dict[layer_name].output
        # ÈÄöËøá‰ªÖÊ∂âÂèäÊçüÂ§±‰∏≠ÁöÑÈùûËæπÁïåÂÉèÁ¥†Êù•ÈÅøÂÖçËæπÁïå‰º™ÂΩ±„ÄÇ
        scaling = K.prod(K.cast(K.shape(activation), 'float32'))
        loss += coeff * K.sum(K.square(activation[:, :, :, :])) / scaling
    
    dream = model.input
    # ËÆ°ÁÆóÊçüÂ§±ÁöÑÊ¢ØÂ∫¶ÁöÑÂèòÂåñ
    grads = K.gradients(loss, dream)[0]
    # Ê†áÂáÜÂåñÊ¢ØÂ∫¶
    grads /= K.maximum(K.mean(K.abs(grads)), 1e-7)
    # ËÆæÁΩÆÂáΩÊï∞‰ª•Ê£ÄÁ¥¢ÁªôÂÆöËæìÂÖ•ÂõæÂÉèÁöÑÊçüÂ§±ÂíåÊ¢ØÂ∫¶ÂÄº„ÄÇ
    outputs = [loss, grads]
    fetch_loss_and_grads = K.function([dream], outputs)
    def eval_loss_and_grads(x):
        outs = fetch_loss_and_grads([x])
        loss_value = outs[0]
        grad_values = outs[1]
        return loss_value, grad_values
    for i in range(epochs):
        loss_value, grad_values = eval_loss_and_grads(img)
        if max_threshold is not None and loss_value > max_threshold:
            break
        img = img + learning_rate * grad_values
    return img
    #********* End *********#
